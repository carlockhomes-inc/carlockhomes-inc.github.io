/* ========================================= M-LAB CARD MAKER - ULTIMATE STABLE 途切れ・エラーをすべて修正した【最終完成版】 ========================================= */ function onOpen() { SpreadsheetApp.getUi() .createMenu('✨ M-LABカード作成') .addItem('ツールを起動', 'showMagicUi') .addToUi(); } /** シート名一覧を取得 */ function getSheetNames() { return SpreadsheetApp.getActiveSpreadsheet().getSheets().map(s => s.getName()); } /** 指定シートのメッセージデータを取得 */ function getSheetData(sheetName) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = sheetName ? ss.getSheetByName(sheetName) : ss.getActiveSheet(); return sheet.getDataRange().getValues().filter(r => r[0] && r[0].toString().trim() !== ""); } catch(e) { return []; } } /** 背景画像をドライブに永久保存 */ function saveBgData(key, base64) { try { const folderName = "M-LAB_System_Assets"; const folders = DriveApp.getFoldersByName(folderName); const folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName); // 既存ファイルを削除 const oldId = PropertiesService.getUserProperties().getProperty(key); if (oldId) { try { DriveApp.getFileById(oldId).setTrashed(true); } catch(e) {} } // 新規保存 const blob = Utilities.newBlob(Utilities.base64Decode(base64.split(',')[1]), "image/jpeg", key + ".jpg"); const file = folder.createFile(blob); PropertiesService.getUserProperties().setProperty(key, file.getId()); return true; } catch(e) { return false; } } /** 保存済みの背景を取得 */ function getSavedBackgrounds() { const props = PropertiesService.getUserProperties().getProperties(); const res = { stA: "", stB: "", stC: "" }; ["stA", "stB", "stC"].forEach(k => { if (props[k]) { try { const b = DriveApp.getFileById(props[k]).getBlob(); res[k] = "data:image/jpeg;base64," + Utilities.base64Encode(b.getBytes()); } catch(e) {} } }); return res; } /** メイン画面を表示 */ function showMagicUi() { const html = `
M-LAB PREMIUM SYSTEM
背景 A
背景 B
背景 C
`; const out = HtmlService.createHtmlOutput(html).setWidth(1100).setHeight(850); SpreadsheetApp.getUi().showModelessDialog(out, 'M-LAB Message Card Hub'); }
